#include <SPI.h>
#include <LoRa.h>

// LoRa and SPI pins for ESP32-S2
#define LORA_SS    5
#define LORA_RST   4
#define LORA_DIO0  2
#define LORA_SCK   18
#define LORA_MISO  19
#define LORA_MOSI  21

String incomingData = "";

void setup() {
  Serial.begin(115200);
  while (!Serial);

  Serial.println("LoRa Transmitter (ESP32-S2)");

  // Initialize SPI bus
  SPI.begin(LORA_SCK, LORA_MISO, LORA_MOSI, LORA_SS);
  LoRa.setPins(LORA_SS, LORA_RST, LORA_DIO0);

  // Initialize LoRa
  if (!LoRa.begin(433E6)) {
    Serial.println("LoRa initialization failed!");
    while (1);
  }

  // Set sync word to match receiver
  LoRa.setSyncWord(0xA5);
  Serial.println("LoRa ready! Waiting for data from Serial...");
}

void loop() {
  // Check for incoming serial data
  if (Serial.available()) {
    incomingData = Serial.readStringUntil('\n');
    incomingData.trim(); // Remove whitespace
    
    if (incomingData.length() > 0) {
      Serial.print("Sending: ");
      Serial.println(incomingData);
      
      // Send packet via LoRa
      LoRa.beginPacket();
      LoRa.print(incomingData);
      LoRa.endPacket();
      
      Serial.println("Packet transmitted!");
    }
  }
}
